Q:Write a program to generate fibonacci series upto a limit


def fibonacci():
    limit = int(input("Enter the limit for the Fibonacci series: "))
    
    a = 0
    b = 1
    print("Fibonacci series up to", limit, ":")
    
    while a <= limit:
        print(a)
        a = b
        b = a + b


fibonacci()
