Q:Write a program to count number of vowels, consonants, digits and special characters using function


def count(s):
    v = 0
    c = 0
    d = 0
    sc = 0
    vowels = "aeiouAEIOU"
    
    for ch in s:
        if ch.isdigit():
            d += 1
        elif ch.isalpha():
            if ch in vowels:
                v += 1
            else:
                c += 1
        else:
            sc += 1
    
    print("Vowels:", v)
    print("Consonants:", c)
    print("Digits:", d)
    print("Special Characters:", sc)
          

str = input("Enter a string:")
count(str)
