Q:Write a program to check a list of integers if the number is divisible by 7 then double it else chabge by thrice the value


def modify_list(numbers):
    changed_list = []  

    for num in numbers:
        if num % 7 == 0:
            x=num*2
            changed_list.append(x)
        else:
            y=num*3
            changed_list.append(y)

    return changed_list
l=eval(input("Enter a list of integers"))

result = modify_list(l)

print("Modified list:", result)
